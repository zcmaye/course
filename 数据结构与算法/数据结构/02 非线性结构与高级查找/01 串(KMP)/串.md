# 5 大字符串匹配算法**超清晰总结**

（面试、考试、写代码都够用，**通俗易懂版**）

我给你整理成**最容易记、最容易讲、最容易理解**的版本，一次性讲透 **BF / RK / KMP / BM / Sunday**。

------

# 一、一句话记住所有算法

1. **BF**：暴力逐个比，不行就回退
2. **RK**：用哈希值代替字符比较
3. **KMP**：利用前缀后缀，**不回退主串**
4. **BM**：从后往前比，**坏字符 + 好后缀**跳得最远
5. **Sunday**：只看窗口下一个字符，**跳得最快、代码最简单**

------

# 二、详细介绍（最易懂版）

## 1. BF 暴力匹配（Brute Force）

**最简单、最慢**

- 思想：主串、模式串一个一个字符比对
- 不匹配：主串回退一位，模式串回到开头
- 时间复杂度：**O(n\*m)**
- 优点：代码最简单
- 缺点：长文本效率极低

------

## 2. RK 哈希匹配（Rabin-Karp）

**用哈希代替逐个比较**

- 思想：计算模式串哈希，再计算主串窗口哈希
- 哈希相同 → 认为匹配（可二次校验防冲突）
- 用**滚动哈希**快速更新
- 时间复杂度：**平均 O (n+m)**
- 优点：比暴力快很多，适合多模式匹配
- 缺点：可能哈希冲突

------

## 3. KMP 算法（教科书经典）

**不回退主串，稳定线性**

- 思想：匹配失败时，利用**最长相等前缀后缀**，模式串直接跳到合适位置
- 关键：**next 数组**
- 时间复杂度：**O(n+m)**
- 优点：稳定、无冲突、必考
- 缺点：理解稍难

------

## 4. BM 算法（工业最快）

**从后往前匹配 + 双规则跳步**

- 思想：
  - **坏字符规则**：不匹配字符没出现 → 跳整个模式串
  - **好后缀规则**：已匹配后缀再出现 → 跳到对应位置
- 每次跳**最大步长**
- 时间复杂度：**最好 O (n/m)，平均 O (n)**
- 优点：**实际最快**，grep / 编辑器都用它
- 缺点：实现最复杂

------

## 5. Sunday 算法（现代最优、最简单）

**只看窗口下一个字符，超快超短**

- 思想：

  

  匹配失败 → 看

  主串窗口下一个字符

  - 不在模式串：直接跳 **m+1**
  - 在模式串：跳到模式串**最右边该字符**对齐

- 时间复杂度：**平均 O (n)**

- 优点：**代码最短、速度不输 BM、最好用**

- 缺点：最坏情况慢（极少遇到）

------

# 四、最终总结（背会就能面试）

1. **BF**：暴力，最简单，最慢
2. **RK**：哈希，快一点
3. **KMP**：稳定线性，面试必考
4. **BM**：实际最快，工程标准
5. **Sunday**：**代码最短、速度最快、最好用**

```cpp
#include <stdio.h>
#include <string.h>

#define BASE 91138
#define MOD 1000000007
#define MAX_CHAR 256
#define MAX_LEN 1000

// ===================== 1. BF 暴力匹配 =====================
int BF(char *txt, char *pat)
{
    int n = strlen(txt), m = strlen(pat);
    int i = 0, j = 0;
    while (i < n && j < m)
    {
        if (txt[i] == pat[j])
        {
            i++; j++;
        }
        else
        {
            i = i - j + 1;
            j = 0;
        }
    }
    return j == m ? (i - j) : -1;
}

// ===================== 2. RK 哈希匹配 =====================
long long getHash(char *s, int len)
{
    long long res = 0;
    for (int i = 0; i < len; i++)
        res = (res * BASE + s[i]) % MOD;
    return res;
}

int RK(char *txt, char *pat)
{
    int n = strlen(txt), m = strlen(pat);
    if (m > n) return -1;
    long long ph = getHash(pat, m);
    long long th = getHash(txt, m);
    long long powB = 1;
    for (int i = 0; i < m - 1; i++)
        powB = (powB * BASE) % MOD;
    if (th == ph) return 0;
    for (int i = 1; i <= n - m; i++)
    {
        th = ((th - txt[i-1] * powB % MOD) % MOD + MOD) % MOD;
        th = (th * BASE + txt[i + m - 1]) % MOD;
        if (th == ph) return i;
    }
    return -1;
}

// ===================== 3. KMP 匹配 =====================
void getNext(char *pat, int *next)
{
    int m = strlen(pat);
    int j = 0;
    next[0] = 0;
    for (int i = 1; i < m; i++)
    {
        while (j > 0 && pat[i] != pat[j])
            j = next[j - 1];
        if (pat[i] == pat[j]) j++;
        next[i] = j;
    }
}

int KMP(char *txt, char *pat)
{
    int n = strlen(txt), m = strlen(pat);
    int next[MAX_LEN];
    getNext(pat, next);
    int j = 0;
    for (int i = 0; i < n; i++)
    {
        while (j > 0 && txt[i] != pat[j])
            j = next[j - 1];
        if (txt[i] == pat[j]) j++;
        if (j == m) return i - m + 1;
    }
    return -1;
}

// ===================== 4. 完整版BM(坏字符+好后缀) =====================
void badChar(char *pat, int m, int bc[])
{
    for (int i = 0; i < MAX_CHAR; i++) bc[i] = m;
    for (int i = 0; i < m; i++) bc[(unsigned char)pat[i]] = m - 1 - i;
}

int maxSuffix(char *pat, int m, int p)
{
    int len = 0;
    int start = p - 1;
    while (start >= 0 && pat[start] == pat[m - 1 - len])
    {
        start--; len++;
    }
    return len;
}

void goodSuffix(char *pat, int m, int gs[])
{
    int i, p;
    for (i = 0; i < m; i++) gs[i] = m;
    for (p = m - 1; p > 0; p--)
    {
        int len = maxSuffix(pat, m, p);
        int j = p - 1 - len;
        if (j >= 0) gs[m - 1 - len] = m - 1 - j;
    }
    int last = 0;
    int maxL = maxSuffix(pat, m, m);
    for (i = m - maxL; i < m; i++)
        if (maxSuffix(pat, m, i + 1) == i + 1)
            last = i + 1;
    for (i = 0; i < m - last; i++)
        gs[i] = m - last;
}

int BM_FULL(char *txt, char *pat)
{
    int n = strlen(txt), m = strlen(pat);
    if (m == 0 || m > n) return -1;
    int bc[MAX_CHAR], gs[MAX_LEN];
    badChar(pat, m, bc);
    goodSuffix(pat, m, gs);
    int i = m - 1;
    while (i < n)
    {
        int j = m - 1;
        while (j >= 0 && txt[i] == pat[j])
        {
            i--; j--;
        }
        if (j < 0) return i + 1;
        int step = bc[(unsigned char)txt[i]];
        if (gs[j] > step) step = gs[j];
        i += step;
    }
    return -1;
}

// ===================== 5. Sunday 算法（超简洁高效） =====================
// 构建偏移表：字符在模式串最右侧到末尾距离
void sundayShift(char *pat, int m, int shift[])
{
    for (int i = 0; i < MAX_CHAR; i++)
        shift[i] = m + 1;
    for (int i = 0; i < m; i++)
        shift[(unsigned char)pat[i]] = m - i;
}

int Sunday(char *txt, char *pat)
{
    int n = strlen(txt), m = strlen(pat);
    if (m > n) return -1;
    int shift[MAX_CHAR];
    sundayShift(pat, m, shift);

    int i = 0;
    while (i <= n - m)
    {
        int j = 0;
        while (j < m && txt[i + j] == pat[j])
            j++;
        if (j == m) return i;
        // 看主串中 匹配窗口下一位字符
        char c = txt[i + m];
        i += shift[(unsigned char)c];
    }
    return -1;
}

// ===================== 主函数统一测试 =====================
int main()
{
    char text[] = "helloabcworldabc";
    char pattern[] = "abc";

    printf("主串: %s\n模式串: %s\n\n", text, pattern);
    printf("BF      : %d\n", BF(text, pattern));
    printf("RK      : %d\n", RK(text, pattern));
    printf("KMP     : %d\n", KMP(text, pattern));
    printf("BM完整版: %d\n", BM_FULL(text, pattern));
    printf("Sunday  : %d\n", Sunday(text, pattern));

    return 0;
}
```
